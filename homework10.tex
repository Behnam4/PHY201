    \begin{equation*}
      a. ~~ \dfrac{dy}{dx}+2xy^2=0
    \end{equation*}
    \begin{equation*}
      b. ~~ \dfrac{y^{\prime}}{y+1}=3
    \end{equation*}
    \begin{equation*}
      c. ~~ y^{\prime}+3y=e^{2x}
    \end{equation*}
    \begin{equation*}
      d. ~~ y^{\prime}=\dfrac{y}{x}
    \end{equation*}
    \begin{equation*}
      e. ~~ e^{x+y}y^{\prime}=x
    \end{equation*}
    \begin{equation*}
      f. ~~ (x+y)y^{\prime}=x-y
    \end{equation*}
    \begin{equation*}
      g. ~~ \dfrac{y^{\prime}-y}{x}=\dfrac{1}{1+x^2}
    \end{equation*}

    \item Prove that an equation of the form:
    $$
    \diff{y}{x} + y P(x) = 0
    $$

    \begin{equation*}
      a. ~~ (2y+3x)dy+(3x^2+3y)dx=0, ~~~ y(0)=1

    \begin{equation*}
      b. ~~ (3x^2y^2+e^y)dy+2(xy^3+1)dx=0, ~~~ y(0)=2

      a. ~~ 2xdy+y(1+x)dx=0
    \end{equation*}
    \begin{equation*}
      b. ~~ cos(x-y)y^{\prime}-cos(x-y)-sin(x-y)=0
    \end{equation*}
    \begin{equation*}
      c. ~~  (y+e^x)y^{\prime}+2ye^x+\dfrac{y^2}{2}=0

    \begin{equation*}
